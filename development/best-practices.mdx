## Project Structure

AgroVetsApp follows a well-organized directory structure to maintain code clarity and scalability.

### Directory Organization

```
AgroVetsApp/
├── App.tsx                 # Entry point, mounts AppNavigator
├── src/
│   ├── navigation/         # All navigation logic
│   │   ├── AppNavigator.tsx
│   │   ├── StackNavigator.tsx
│   │   ├── TabNavigator.tsx
│   │   ├── DrawerNavigator.tsx

│   │   ├── Home/

│   │   ├── ui/            # Generic UI components
│   │   └── common/        # Shared components
│   ├── hooks/             # Custom React hooks
```

## TypeScript Best Practices

### Type Definitions

#### Navigation Types

Define your navigation types in `src/navigation/types.ts`:

```tsx
import type { NativeStackScreenProps } from '@react-navigation/native-stack';
import type { BottomTabScreenProps } from '@react-navigation/bottom-tabs';

// Define your navigation structure
export type RootStackParamList = {
  Home: undefined;
  Profile: { userId: string };
  Settings: { section?: string };
};

export type TabParamList = {
  HomeTab: undefined;
  ProfileTab: undefined;

// Create screen prop types
```

Use these types in your screens:

```tsx
import type { ProfileScreenProps } from '../navigation/types';

export default function ProfileScreen({ route, navigation }: ProfileScreenProps) {
  const { userId } = route.params; // TypeScript knows userId is a string
  
  // navigation is fully typed

  return <View>{/* ... */}</View>;

```

#### API Response Types

Type your API responses in `src/services/api.ts`:

```tsx
// Define response types
interface User {
  id: string;
  name: string;
  email: string;
  createdAt: string;
}

interface ApiResponse<T> {
  data: T;
  success: boolean;
  message?: string;

// Usage
```

### Avoid Common TypeScript Pitfalls

<Accordion title="Don't use 'any' type">
  **Bad**:
  ```tsx
  function handleData(data: any) {
    return data.value; // No type safety
  }
  ```

  **Good**:
  ```tsx
  interface Data {
    value: string;
  }

  function handleData(data: Data) {
    return data.value; // Type-safe
  }

  // Or use 'unknown' if type is truly unknown
  function handleData(data: unknown) {
    if (typeof data === 'object' && data !== null && 'value' in data) {
      return (data as Data).value;
    }
  }
  ```
</Accordion>

<Accordion title="Use type inference when possible">
  **Over-typed**:
  ```tsx
  const name: string = 'John';
  const count: number = 5;
  const items: string[] = ['a', 'b', 'c'];
  ```

  **Better**:
  ```tsx
  const name = 'John'; // TypeScript infers string
  const count = 5; // TypeScript infers number
  const items = ['a', 'b', 'c']; // TypeScript infers string[]
  ```

  **But do type complex objects**:
  ```tsx
  interface UserSettings {
    theme: 'light' | 'dark';
    notifications: boolean;
  }

  const settings: UserSettings = {
    theme: 'light',
    notifications: true,
  };
  ```
</Accordion>

<Accordion title="Properly handle null and undefined">
  With strict mode enabled, always check for null/undefined:

  ```tsx
  // Bad
  function getDisplayName(user: User | null) {
    return user.name; // Error: Object is possibly 'null'
  }

  // Good - Use optional chaining
  function getDisplayName(user: User | null) {
    return user?.name ?? 'Guest';
  }

  // Good - Use type guard
  function getDisplayName(user: User | null) {
    if (!user) return 'Guest';
    return user.name;
  }
  ```
</Accordion>

## React and React Native Best Practices

### Component Design

<Accordion title="Keep components small and focused">
  Each component should have a single responsibility:

  ```tsx
  // Bad - Too many responsibilities
  function ProfileScreen() {
    const [user, setUser] = useState(null);
    const [posts, setPosts] = useState([]);
    const [loading, setLoading] = useState(true);
    
    // Fetching logic
    useEffect(() => { /* ... */ }, []);
    
    return (
      <View>
        {/* Header */}
        {/* Avatar */}
        {/* Bio */}
        {/* Stats */}
        {/* Posts list */}
      </View>
    );
  }

  // Good - Split into focused components
  function ProfileScreen() {
    const { user, loading } = useProfile();
    
    if (loading) return <LoadingSpinner />;
    if (!user) return <ErrorMessage />;
    
    return (
      <View>
        <ProfileHeader user={user} />
        <ProfileBio bio={user.bio} />
        <ProfileStats stats={user.stats} />
        <ProfilePosts userId={user.id} />
      </View>
    );
  }
  ```
</Accordion>

<Accordion title="Use custom hooks for logic reuse">
  Extract reusable logic into custom hooks in `src/hooks/`:

  ```tsx
  // src/hooks/useProfile.ts
  export function useProfile(userId: string) {
    const [user, setUser] = useState<User | null>(null);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState<Error | null>(null);

    useEffect(() => {
      async function fetchUser() {
        try {
          const data = await api.getUser(userId);
          setUser(data);
        } catch (err) {
          setError(err as Error);
        } finally {
          setLoading(false);
        }
      }
      fetchUser();
    }, [userId]);

    return { user, loading, error };
  }

  // Usage in component
  function ProfileScreen({ route }: ProfileScreenProps) {
    const { user, loading, error } = useProfile(route.params.userId);
    // ...
  }
  ```
</Accordion>

## State Management

### Use Context for Global State

Create contexts in `src/contexts/`:

```tsx
// src/contexts/AuthContext.tsx
import React, { createContext, useContext, useState } from 'react';

interface AuthContextType {
  user: User | null;
  login: (email: string, password: string) => Promise<void>;
  logout: () => void;
}

const AuthContext = createContext<AuthContextType | undefined>(undefined);

export function AuthProvider({ children }: { children: React.ReactNode }) {
  const [user, setUser] = useState<User | null>(null);

  const login = async (email: string, password: string) => {
    const userData = await api.login(email, password);
    setUser(userData);
  };

  const logout = () => setUser(null);

  return (
    <AuthContext.Provider value={{ user, login, logout }}>
      {children}
    </AuthContext.Provider>
  );
}

export function useAuth() {
  const context = useContext(AuthContext);
  if (!context) {
    throw new Error('useAuth must be used within AuthProvider');
  }
  return context;
}
```

Then wrap your app in `App.tsx`:

```tsx
import { AuthProvider } from './src/contexts/AuthContext';

export default function App() {
  return (
    <AuthProvider>
      <AppNavigator />
    </AuthProvider>
  );
}
```
